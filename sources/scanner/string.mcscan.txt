@def any-character: .
@def unknown: ?

@comment: token <string> handle of schema <">(. except of <\n>)<">
@comment: token <word> handle of schema ([a-z] not empty)

rule parse:
  follow parse_string
  follow parse_comment
  follow parse_word

rule parse_word
  layout: 
    "<:result>(.+)", '<:result>(.+)'
  work(result):
    push <word>, result 
end

rule parse_string:
  schema <">(result as . except of <\n>)<">
  push <string>, result

rule parse_comment:
  schema <//>(result as . except of <\n>)<\n>
  push <comment>, result

rule main:
  follow parse
